 void displayText(int n)

 int CountEquels(char** Strings, int n) 
 {
	 int count = 0;
	 for (int i = 0; i < n; i++) {
		 for (int j = i + 1; j < n - 1; j++) 
		 {
			 if (!strcmp(Strings[i], Strings[j])) 
			 {
				 count++;
			 }
		 }
	 }
	 return count;
 }
